def get_day_month(dt):
    return dt.day

data["dom"]=data["Date/Time"].map(get_day_month)

def get_weekday(dt):
    return dt.weekday()
data["weekday"]=data["Date/Time"].map(get_weekday)

def get_hour(dt):
    return dt.hour
data["hour"]=data['Date/Time'].map(get_hour)
